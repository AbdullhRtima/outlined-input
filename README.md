# react-native-outlined-input

## Install

```bash
npm install --save react-native-outlined-input
```

```tsx

import{ OutlinedInput } from 'react-native-outlined-input'
import 'react-native-outlined-input/dist/index.css'

const App =()=> {
    // react-native-outlined-input
    
    return <OutlinedInput />

}
```
